% Define a custom tcolorbox for proofs
\newtcolorbox{myproof}{
  enhanced jigsaw,
  sharp corners,
  colframe=proofpurple,
  colback=proofpurple!0,
  borderline west={1pt}{0pt}{proofpurple},
  borderline south={1pt}{0pt}{proofpurple},
  before skip=10pt,
  after skip=10pt,
  boxrule=0pt,
  left=10pt,
  right=10pt,
  breakable,
}

% Redefine the proof environment to use the custom tcolorbox
\renewenvironment{proof}[1][\proofname]{
  \begin{myproof}
    \textbf{#1.}
}{\hfill$\qed$\end{myproof}
}
